model{
	# Variable que decide qué moneda está cargada
	for(i in 1:m){
	p[i] <- 1/m
}	
	c ~ dcat(p[])  
	
	#Priors de las variables.

	
	
    # La distribución depende del valor de k
	
    for (i in 1:m){
        Thetau[i] ~ dbeta(1, 1)
		Theta[i] <- ifelse(c == i, Thetau[i], 1/2)  
    }

	for (i in 1:m){
		k[i] ~ dbin(Theta[i], n)
	}
}
